    function generatePremiumArena(screenNumber, rows, columns) {
        selectedScreenNum = screenNumber;
        document.getElementById('screen-selector-modal').style.display = 'none';
        document.getElementById('m-name').innerText = `${currentMovie} 🍿 SCREEN ${screenNumber}`;
        
        const grid = document.getElementById('s-grid');
        grid.innerHTML = "";
        updatePriceBoard(0, []);

        let centerAisleIndex = Math.floor(columns / 2);
        grid.style.gridTemplateColumns = `repeat(${columns + 1}, auto)`;
        
        for(let r = 0; r < rows; r++) {
            let rowLetter = ROW_LETTERS[r] || `R${r}`;
            
            for(let cNum = 1; cNum <= columns; cNum++) {
                if(cNum === centerAisleIndex + 1) {
                    let walkwayGap = document.createElement('div');
                    walkwayGap.className = 'seat aisle-gap';
                    grid.appendChild(walkwayGap);
                }

                let seatIdentifier = `${rowLetter}-${cNum}`;
                let s = document.createElement('div');
                s.className = 'seat';
                s.innerText = cNum;
                s.setAttribute('data-seatid', seatIdentifier);
                
                if(Math.random() < 0.22) {
                    s.classList.add('booked-already');
                }
                
                s.addEventListener('mousedown', () => {
                    if(!s.classList.contains('booked-already')) {
                        s.classList.toggle('selected');
                        refreshActiveCalculation();
                    }
                });

                s.addEventListener('mouseenter', () => {
                    if(isMouseDown && !s.classList.contains('booked-already')) {
                        s.classList.add('selected');
                        refreshActiveCalculation();
                    }
                });

                grid.appendChild(s);
            }
        }
        document.getElementById('booking-modal').style.display = 'flex';
    }

    function refreshActiveCalculation() {
        const selectedChairs = document.querySelectorAll('.seat.selected');
        let selectedSeatNamesArray = [];
        selectedChairs.forEach(chair => {
            selectedSeatNamesArray.push(chair.getAttribute('data-seatid'));
        });
        updatePriceBoard(selectedChairs.length, selectedSeatNamesArray);
    }

    function updatePriceBoard(selectedCount, seatNamesArray) {
        billCalculation.seats = selectedCount;
        billCalculation.seatNames = seatNamesArray.join(', ');
        
        if(selectedCount === 0) {
            billCalculation.base = 0; billCalculation.handling = 0; billCalculation.gst = 0; billCalculation.total = 0;
            document.getElementById('live-seat-names').innerText = "None";
        } else {
            billCalculation.base = selectedCount * TICKET_RATE;
            billCalculation.handling = HANDLING_CHARGE;
            billCalculation.gst = (billCalculation.base + billCalculation.handling) * 0.18;
            billCalculation.total = billCalculation.base + billCalculation.handling + billCalculation.gst;
            document.getElementById('live-seat-names').innerText = billCalculation.seatNames;
        }
        
        document.getElementById('live-count').innerText = billCalculation.seats;
        document.getElementById('live-base').innerText = "₹" + billCalculation.base.toFixed(2);
        document.getElementById('live-handling').innerText = "₹" + billCalculation.handling.toFixed(2);
        document.getElementById('live-gst').innerText = "₹" + billCalculation.gst.toFixed(2);
        document.getElementById('live-total').innerText = "₹" + billCalculation.total.toFixed(2);
    }
