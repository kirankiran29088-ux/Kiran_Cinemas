<script>
    let currentMovie = "";
    function startBooking(name, count) {
        currentMovie = name;
        document.getElementById('m-name').innerText = name + " - " + count + " Seats";
        const grid = document.getElementById('s-grid');
        grid.innerHTML = "";
        
        // Accurate Column Logic for 2000/1000 seats
        grid.style.gridTemplateColumns = (count === 2000) ? "repeat(50, 1fr)" : "repeat(40, 1fr)";
        
        // We show 800-1000 visual seats for smooth performance
        let displayLimit = (count === 2000) ? 1000 : 600; 
        for(let i=1; i<=displayLimit; i++) {
            let s = document.createElement('div');
            s.className = 'seat';
            if(Math.random() < 0.1) s.classList.add('booked-already');
            s.onclick = () => { if(!s.classList.contains('booked-already')) s.classList.toggle('selected'); };
            grid.appendChild(s);
        }
        document.getElementById('booking-modal').style.display = 'flex';
    }

    function confirmBooking() {
        const selected = document.querySelectorAll('.seat.selected').length;
        if(selected === 0) return alert("Please select your seats, bro!");
        document.getElementById('final-movie').innerText = currentMovie;
        document.getElementById('final-count').innerText = selected;
        document.getElementById('confirm-page').style.display = 'flex';
    }
</script>
